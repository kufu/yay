def create
  user = User.find_by(email: session_params[:email])

  if user&.authenticate(session_params[:password])
    session[:user_id] = user.id
    redirect_to root_url, notice: "ログインしました。"
  else
    render :new, status: :unprocessable_entity
  end
end
